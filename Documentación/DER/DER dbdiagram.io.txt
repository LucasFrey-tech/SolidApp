Table cuenta {
  id int [pk, increment]
  correo varchar(255) [unique, not null]
  clave varchar(255) [not null]
  role varchar(20) [not null]
  calle varchar(80)
  numero varchar(10)
  codigo_postal varchar(10)
  ciudad varchar(50)
  provincia varchar(50)
  prefijo varchar(5)
  telefono varchar(20)
  deshabilitado bit [not null, default: 0]
  verificada bit [not null, default: 0]
  fecha_registro datetime2 [not null, default: `getdate()`]
  ultimo_cambio datetime2 [not null, default: `getdate()`]
  ultima_conexion datetime2
  resetPasswordToken varchar(255)
  resetPasswordExpires datetime2
}

Table perfil_usuario {
  id int [pk, increment]
  documento varchar(15) [unique, not null]
  nombre varchar(50) [not null]
  apellido varchar(50) [not null]
  puntos int [not null, default: 0]
  cuentaId int [ref: - cuenta.id]
  departamento varchar(10)
}

Table perfil_empresa {
  id int [pk, increment]
  cuit varchar(20) [unique, not null]
  razon_social varchar(100) [not null]
  rubro varchar(50) [not null, default: '']
  nombre_empresa varchar(15) [not null]
  verificada bit [not null, default: 0]
  logo varchar(255)
  web varchar(255)
  cuentaId int [ref: - cuenta.id]
  descripcion varchar(500) [not null, default: '']
}

Table perfil_organizacion {
  id int [pk, increment]
  cuit varchar(20) [unique, not null]
  razon_social varchar(100) [not null]
  nombre_organizacion varchar(100) [not null]
  descripcion varchar(255) [not null, default: '']
  verificada bit [not null, default: 0]
  web varchar(255) [not null, default: '']
  cuentaId int [ref: - cuenta.id]
}

Table beneficios {
  id int [pk, increment]
  titulo varchar(255) [not null]
  tipo varchar(50) [not null]
  detalle varchar(255) [not null]
  cantidad int [not null]
  fecha_registro datetime2 [not null, default: `getdate()`]
  ultimo_cambio datetime2 [not null, default: `getdate()`]
  valor int [not null]
  estado varchar(20) [not null, default: 'PENDIENTE']
  id_empresa int [ref: > perfil_empresa.id]
}

Table campañas {
  id int [pk, increment]
  estado varchar(20) [not null]
  titulo varchar(255) [not null]
  descripcion varchar(255) [not null]
  fecha_Inicio date [not null]
  fecha_Fin date [not null]
  fecha_Registro datetime2 [not null, default: `getdate()`]
  objetivo int [not null, default: 0]
  puntos int [not null, default: 0]
  ultimo_cambio datetime2 [not null, default: `getdate()`]
  id_organizacion int [ref: > perfil_organizacion.id]
}

Table donaciones {
  id int [pk, increment]
  titulo varchar(100) [not null]
  detalle varchar(255) [not null]
  tipo varchar(25) [not null, default: 'Articulo']
  cantidad int [not null]
  estado int [not null, default: 0]
  puntos int
  fecha_registro datetime2 [not null, default: `getdate()`]
  fecha_estado datetime2
  motivo_rechazo varchar(255)
  id_campaña int [ref: > campañas.id]
  usuarioId int [ref: > perfil_usuario.id]
}

Table usuarios_beneficios {
  id int [pk, increment]
  cantidad int [not null, default: 1]
  usados int [not null, default: 0]
  fecha_reclamo datetime2 [not null, default: `getdate()`]
  fecha_uso datetime2
  ultimo_cambio datetime2 [not null, default: `getdate()`]
  id_usuario int [ref: > perfil_usuario.id]
  id_beneficio int [ref: > beneficios.id]
  estado varchar(255) [not null, default: 'ACTIVO']
}

Table ranking_donador {
  id_usuario int [pk, ref: - perfil_usuario.id]
  puntos int [not null]
}

Table imagenes_beneficio {
  id int [pk, increment]
  imagen varchar(255) [not null]
  id_beneficio int [ref: > beneficios.id]
}

Table imagenes_campaña {
  id int [pk, increment]
  imagen varchar(255) [not null]
  esPortada bit [not null, default: 0]
  campañas_id int [ref: > campañas.id]
}

Table imagenes_donacion {
  id int [pk, increment]
  imagen varchar(255) [not null]
  donaciones_id int [ref: > donaciones.id]
}

Table imagenes_organizacion {
  id int [pk, increment]
  logo varchar(255) [not null]
  banner varchar(255) [not null]
  organizaciones_id int [ref: > perfil_organizacion.id]
}
